### **Code Implementation**  

<CodeBlock language="c">
{
`#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Structure for Student
struct Student {
    int roll;
    char name[50];
    float marks;
    struct Student *next;
};

// Function to add a student
struct Student* addStudent(struct Student *head, int roll, char name[], float marks) {
    struct Student *newStudent = (struct Student*)malloc(sizeof(struct Student));
    newStudent->roll = roll;
    strcpy(newStudent->name, name);
    newStudent->marks = marks;
    newStudent->next = head;
    return newStudent;
}

// Function to display students
void displayStudents(struct Student *head) {
    struct Student *temp = head;
    while (temp != NULL) {
        printf("Roll: %d, Name: %s, Marks: %.2f\\n", temp->roll, temp->name, temp->marks);
        temp = temp->next;
    }
}

// Function to delete a student
struct Student* deleteStudent(struct Student *head, int roll) {
    struct Student *temp = head, *prev = NULL;
    while (temp != NULL && temp->roll != roll) {
        prev = temp;
        temp = temp->next;
    }
    if (temp == NULL) return head;
    if (prev == NULL) head = temp->next;
    else prev->next = temp->next;
    free(temp);
    return head;
}

// Main function
int main() {
    struct Student *head = NULL;

    // Adding students
    head = addStudent(head, 101, "Alice", 85.5);
    head = addStudent(head, 102, "Bob", 90.0);
    head = addStudent(head, 103, "Charlie", 78.0);

    printf("Student Records:\\n");
    displayStudents(head);

    // Deleting a student
    head = deleteStudent(head, 102);
    printf("\\nAfter Deletion:\\n");
    displayStudents(head);

    return 0;
}`
}

    // Performing calculation
    result = calculate(num1, num2, op);
    printf("Result: %.2f\\n", result);

    return 0;
}`
